void Sieve(int n) {
	ip[1] = 1; c[1] = 0;
	for(int i = 2; i <= n; i++) {
		if(!ip[i]) {
			pr[++tot] = i; c[i] = 1;
		}
		for(int j = 1; j <= tot && pr[j] <= n / i; j++) {

		}
	}
